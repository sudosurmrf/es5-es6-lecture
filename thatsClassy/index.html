ES5 vs ES6: Constructor Functions vs Classes

ES5 Constructor Function

function Person1(name, age) {
  this.name = name;
  this.age = age;
}

Person1.prototype.greet = function() {
  return 'Hello ' + this.name;
};

ES6 Class

class Person {
  constructor(name, age) {
      this.name = name;
      this.age = age;
  }

  greet() {
      return `Hello ${this.name}`;
  }
}
