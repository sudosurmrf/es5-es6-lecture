Hoist the Sails: Function Hoisting - ES5 vs ES6

Arrr! Let's set sail and see how hoisting works in ES5 and ES6!

Pirate's Log:

Function Code Snippets:

ES5 Function Declaration (Hoisted)


logToScreen("Trying to call the ES5 function before its declaration:");
  try {
    es5Function();
  } catch (error) {
    logToScreen("Error: " + error.message, true);
}  

function es5Function() {
    console.log("Ahoy! This ES5 function be hoisted to the top, ready for action!");
}

ES6 Function Expression (not Hoisted)


logToScreen("Calling the ES6 function expression before its declaration:");
  try {
    es6Function(); 
  } catch (error) {
    logToScreen("Arrr! Error: " + error.message, true);
} 

let es6Function = function() {
    console.log("Arrr! This ES6 function be stuck, no hoisting here.");
};

ES6 Arrow Function (Not Hoisted)


logToScreen("Calling the ES6 arrow function before its declaration:");
  try {
    es6ArrowFunction();
  } catch (error) {
    logToScreen("Shiver me timbers! Error: " + error.message, true);
}

const es6ArrowFunction = () => {
    console.log("This ES6 arrow function stays anchored, no hoisting in sight!");
};
