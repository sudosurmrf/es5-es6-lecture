ES5 vs ES6: Import/Export Modules

This demo explains how ES5 and ES6 handle module imports and exports differently.

ES5 Module Example


// file: es5-module.js
var mathModule = (function() {
  var privateVar = 42;

  function add(x, y) {
    return x + y;
  }

  return {
    add: add,
    answer: privateVar
  };
})();

// file: es5-main.js
var result = mathModule.add(10, 5);
console.log('ES5 Result:', result);  // Output: ES5 Result: 15
console.log('The answer to everything is:', mathModule.answer);

ES6 Module Example


// file: main.js
import { add, answer } from './math.js';
          
const result = add(10, 5);
console.log('ES6 Result:', result);  // Output: ES6 Result: 15
console.log('The answer to everything is:', answer);

// file: math.js
export const answer = 42;

const add = (x, y) => {
  return x + y;
}
